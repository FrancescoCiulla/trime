081  private class CandidateSpan extends ClickableSpan{
082      int index;
083      Typeface tf;
084      int hi_text, hi_back, text;
085      public CandidateSpan(int i, Typeface _tf, int _hi_text, int _hi_back, int _text) {
086          super();
087          index = i;
088          tf = _tf;
089          hi_text = _hi_text;
090          hi_back = _hi_back;
091          text = _text;
092      }
093      @Override
094      public void onClick(View tv) {
095         Trime.getService().onPickCandidate(index);
096      }
097      @Override
098      public void updateDrawState(TextPaint ds) {
099          ds.setUnderlineText(false);
100          ds.setTypeface(tf);
101          if (index == highlightIndex) {
102            ds.setColor(hi_text);
103            ds.bgColor = hi_back;
104          } else {
105            ds.setColor(text);
106          }
107      }
108  }

152  @Override
153  public boolean onTouchEvent(MotionEvent event) {
154    int action = event.getAction();
155    if (action == MotionEvent.ACTION_UP) {
156      int n = getOffsetForPosition(event.getX(), event.getY());
157      if (composition_pos[0] <= n && n <= composition_pos[1]) {
158        String s = getText().toString().substring(n, composition_pos[1]).replace(" ", "").replace("‸", "");
159        n = Rime.RimeGetInput().length() - s.length(); //從右側定位
160        Rime.RimeSetCaretPos(n);
161        Trime.getService().updateComposing();
162        return true;
163      }
164    } else if (!movable.contentEquals("false") && (action == MotionEvent.ACTION_MOVE || action == MotionEvent.ACTION_DOWN)) {
165      int n = getOffsetForPosition(event.getX(), event.getY());
166      if (move_pos[0] <= n && n <= move_pos[1]) {
167        if (action == MotionEvent.ACTION_DOWN) {
168          if (first_move || movable.contentEquals("once")) {
169            first_move = false;
170            int location[] = new int[2];
171            getLocationOnScreen(location);
172            mCurrentX =location[0];
173            mCurrentY = location[1];
174          }
175          mDx = mCurrentX - event.getRawX();
176          mDy = mCurrentY - event.getRawY();
177        } else { //MotionEvent.ACTION_MOVE
178          mCurrentX = (int) (event.getRawX() + mDx);
179          mCurrentY = (int) (event.getRawY() + mDy);
180          Trime.getService().updateWindow(mCurrentX, mCurrentY);
181        }
182        return true;
183      }
184    }
185    return super.onTouchEvent(event);
186  }

239  private Object getAlign(Map m) {
240    Layout.Alignment i = Layout.Alignment.ALIGN_NORMAL;
241    if (m.containsKey("align")) {
242      String align = Function.getString(m, "align");
243      switch (align) {
244        case "left":
245        case "normal":
246          i = Layout.Alignment.ALIGN_NORMAL;
247          break;
248        case "right":
249        case "opposite":
250          i = Layout.Alignment.ALIGN_OPPOSITE;
251          break;
252        case "center":
253          i = Layout.Alignment.ALIGN_CENTER;
254          break;
255      }
256    }
257    return new AlignmentSpan.Standard(i);
258  }

260  private void appendComposition(Map m) {
261    Rime.RimeComposition r = Rime.getComposition();
262    String s = r.getText();
263    int start, end;
264    String sep = Function.getString(m, "start");
265    if (!Function.isEmpty(sep)) {
266      start = ss.length();
267      ss.append(sep);
268      end = ss.length();
269      ss.setSpan(getAlign(m), start, end, span);
270    }
271    start = ss.length();
272    ss.append(s);
273    end = ss.length();
274    ss.setSpan(getAlign(m), start, end, span);
275    composition_pos[0] = start;
276    composition_pos[1] = end;
277    ss.setSpan(new CompositionSpan(), start, end, span);
278    ss.setSpan(new AbsoluteSizeSpan(text_size), start, end, span);
279    if (VERSION.SDK_INT >= VERSION_CODES.LOLLIPOP && m.containsKey("letter_spacing")) {
280      Object o = m.get("letter_spacing");
281      double size = 1d;
282      if (o instanceof Integer) size = ((Integer)o).doubleValue();
283      else if (o instanceof Float) size = ((Float)o).doubleValue();
284      else if (o instanceof Double) size = ((Double)o).doubleValue();
285      ss.setSpan(new LetterSpacingSpan((float)size), start, end, span);
286    }
287    start = composition_pos[0] + r.getStart();
288    end = composition_pos[0] + r.getEnd();
289    ss.setSpan(new ForegroundColorSpan(hilited_text_color), start, end, span);
290    ss.setSpan(new BackgroundColorSpan(hilited_back_color), start, end, span);
291    sep = Function.getString(m, "end");
292    if (!Function.isEmpty(sep))ss.append(sep);
293  }

295  private int appendCandidates(Map m, int length) {
296    int start, end;
297    int start_num = 0;
298    Rime.RimeCandidate[] candidates = Rime.getCandidates();
299    if (candidates == null) return start_num;
300    String sep = Function.getString(m, "start");
301    highlightIndex = candidate_use_cursor ? Rime.getCandHighlightIndex() : -1;
302    String label_format = Function.getString(m, "label");
303    String candidate_format = Function.getString(m, "candidate");
304    String comment_format = Function.getString(m, "comment");
305    String line = Function.getString(m, "sep");
306    int last_cand_length = 0;
307    int line_length = 0;
308    String[] labels = Rime.getSelectLabels();
309    int i = -1;
310    candidate_num = 0;
311    for (Rime.RimeCandidate o: candidates) {
312      String cand = o.text;
313      if (Function.isEmpty(cand)) cand = "";
314      i++;
315      if (candidate_num >= max_entries) {
316        if (start_num == 0 && candidate_num == i) start_num = candidate_num;
317        break;
318      }
319      if (cand.length() < length) {
320        if (start_num == 0 && candidate_num == i) start_num = candidate_num;
321        if (all_phrases) continue;
322        else break;
323      }
324      cand = String.format(candidate_format, cand);
325      String line_sep;
326      if (candidate_num == 0) {
327        line_sep = sep;
328      } else if ((sticky_lines > 0 && sticky_lines >= i)
329            || (max_length > 0 && line_length + cand.length() > max_length)){
330        line_sep = "\n";
331        line_length = 0;
332      } else {
333        line_sep = line;
334      }
335      if (!Function.isEmpty(line_sep)) {
336        start = ss.length();
337        ss.append(line_sep);
338        end = ss.length();
339        ss.setSpan(getAlign(m), start, end, span);
340      }
341      if (!Function.isEmpty(label_format) && labels != null) {
342        String label = String.format(label_format, labels[i]);
343        start = ss.length();
344        ss.append(label);
345        end = ss.length();
346        ss.setSpan(new CandidateSpan(i, tfLabel, hilited_candidate_text_color, hilited_candidate_back_color, label_color), start, end, span);
347        ss.setSpan(new AbsoluteSizeSpan(label_text_size), start, end, span);
348      }
349      start = ss.length();
350      ss.append(cand);
351      end = ss.length();
352      line_length += cand.length();
353      ss.setSpan(getAlign(m), start, end, span);
354      ss.setSpan(new CandidateSpan(i, tfCandidate, hilited_candidate_text_color, hilited_candidate_back_color, candidate_text_color), start, end, span);
355      ss.setSpan(new AbsoluteSizeSpan(candidate_text_size), start, end, span);
356      String comment = o.comment;
357      if (show_comment && !Function.isEmpty(comment_format) && !Function.isEmpty(comment)) {
358        comment = String.format(comment_format, comment);
359        start = ss.length();
360        ss.append(comment);
361        end = ss.length();
362        ss.setSpan(getAlign(m), start, end, span);
363        ss.setSpan(new CandidateSpan(i, tfComment, hilited_comment_text_color, hilited_candidate_back_color, comment_text_color), start, end, span);
364        ss.setSpan(new AbsoluteSizeSpan(comment_text_size), start, end, span);
365        line_length += comment.length();
366      }
367      candidate_num++;
368    }
369    if (start_num == 0 && candidate_num == i + 1) start_num = candidate_num;
370    sep = Function.getString(m, "end");
371    if (!Function.isEmpty(sep)) ss.append(sep);
372    return start_num;
373  }

375  private void appendButton(Map m) {
376    if (m.containsKey("when")) {
377      String when = Function.getString(m, "when");
378      if (when.contentEquals("paging") && !Rime.isPaging()) return;
379      if (when.contentEquals("has_menu") && !Rime.hasMenu()) return;
380    }
381    String label;
382    Event e = new Event(null, Function.getString(m, "click"));
383    if (m.containsKey("label")) label = Function.getString(m, "label");
384    else label = e.getLabel();
385    int start, end;
386    String sep = null;
387    if (m.containsKey("start")) sep = Function.getString(m, "start");
388    if (!Function.isEmpty(sep)) {
389      start = ss.length();
390      ss.append(sep);
391      end = ss.length();
392      ss.setSpan(getAlign(m), start, end, span);
393    }
394    start = ss.length();
395    ss.append(label);
396    end = ss.length();
397    ss.setSpan(getAlign(m), start, end, span);
398    ss.setSpan(new EventSpan(e), start, end, span);
399    ss.setSpan(new AbsoluteSizeSpan(key_text_size), start, end, span);
400    sep = Function.getString(m, "end");
401    if (!Function.isEmpty(sep)) ss.append(sep);
402  }

426  public int setWindow(int length) {
427    if (getVisibility() != View.VISIBLE) return 0;
428    Rime.RimeComposition r = Rime.getComposition();
429    if (r == null) return 0;
430    String s = r.getText();
431    if (Function.isEmpty(s)) return 0;
432    setSingleLine(true); //設置單行
433    ss = new SpannableStringBuilder();
434    int start_num = 0;
435    for (Map<String,Object> m: components) {
436      if (m.containsKey("composition")) appendComposition(m);
437      else if (m.containsKey("candidate")) start_num = appendCandidates(m, length);
438      else if (m.containsKey("click"))appendButton(m);
439      else if (m.containsKey("move"))appendMove(m);
440    }
441    if (candidate_num > 0 || ss.toString().contains("\n")) setSingleLine(false); //設置單行
442    setText(ss);
443    setMovementMethod(LinkMovementMethod.getInstance());
444    return start_num;
445  }
